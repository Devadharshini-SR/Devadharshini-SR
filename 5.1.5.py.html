import math num_elements = int(input()) numbers = [] print("Enter the numbers:") for i in range(num_elements): num = float(input(f"Enter number {i + 1}: ")) numbers.append(num) total_sum = 0 for num in numbers: total_sum += num mean = total_sum / num_elements sum_of_squares = 0 for num in numbers: sum_of_squares += (num - mean) ** 2 variance = sum_of_squares / num_elements deviation = math.sqrt(variance) print(f"Mean: {mean}") print(f"Variance: {variance}") print(f"Deviation: {deviation}")
